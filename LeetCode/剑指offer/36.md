//这边是给了一个中序遍历
class Solution {
    //思路跟中序一样 但是是中序的加深操作
    Node pre,head;
    public Node treeToDoublyList(Node root) {
        if(root==null){
            return root;
        }
        dfs(root);
        head.left=pre;
        pre.right=head;
        return head;

    }
    //感觉是写一个中序遍历 里面套递归的写法 参数两个 因为定义了全局变量所以不用使用局部变量了
    public void dfs(Node cur){
        if(cur==null){
            return;
        }
        //这边顺序关系很大，left指针不注意会导致栈溢出问题
        dfs(cur.left);
        cur.left=pre;
        if(cur.left==null){
            head=cur;
        }else{
            pre.right=cur;
        }
        //当前节点的左指针是指向pre
        pre=cur;
        dfs(cur.right);
    }

}
