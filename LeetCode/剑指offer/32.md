class Solution {
    public int[] levelOrder(TreeNode root) {
        //直接层次遍历 list转数组  
        if(root==null){
            return new int []{};
        }
        int []ans=new int[1024];
        int counter=0;
        Queue<TreeNode> queue=new LinkedList<TreeNode>();
        queue.offer(root);
        while(!queue.isEmpty()){
            TreeNode cur=queue.poll();
            ans[counter++]=cur.val;
            if(cur.left!=null){
                queue.offer(cur.left);
            }
            if(cur.right!=null){
                queue.offer(cur.right);
            }
        }
        return Arrays.copyOf(ans,counter);
    }
}
