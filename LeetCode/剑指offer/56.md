class Solution {
    public int[] singleNumbers(int[] nums) {
        //基本只有两种情况1.尾部不同 根据尾部不同进行异或操作结束
        //2. 尾部相同 的情况 如何处理呢


        //思路对一半  首先找到两个不同数字的最后不同的那一位 然后以这个为边界将整个数组分为两个部分
        int sum=0;
        for (int a:nums) {
            sum^=a;
        }
        //获得到两个数不同的第一位
        int mid=sum&(-sum);
        int a=0;
        int b=0;
        for (int aa:nums) {
            //mid 和0 都可以
            if((aa&mid)==mid){
                a^=aa;
            }else{
                b^=aa;
            }
        }
        return new int[]{a,b};
    }
}
