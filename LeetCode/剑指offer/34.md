```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private List<List<Integer>> res;
    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        //简单回溯
        res=new ArrayList<>();
        back(root,sum,new ArrayList<>());
        return res;
    }
    private void back(TreeNode node,int target,List<Integer> collector){
        if(node==null){
            return;
        }
        collector.add(node.val);
        target-=node.val;
        if(target==0&&node.left==null&&node.right==null){
            res.add(new ArrayList<>(collector));
        }
        else{
            back(node.left,target,collector);
            back(node.right,target,collector);
        }
        collector.remove(collector.size()-1);
    }
}
```
