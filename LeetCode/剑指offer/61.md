class Solution {
   public  boolean isStraight(int[] nums) {
        //先判断 0的个数
        Arrays.sort(nums);
        int zeroCount = 0;
        for (int a : nums) {
            if (a == 0) {
                zeroCount++;
            }
        }
        if(zeroCount==0){
            return isS(nums);
        }else{
            //直接计算每个不为0的数字之间的差
           int sum=0;
            for(int i=zeroCount;i<4;i++){
                int a=nums[i+1]-nums[i]-1;
                if(a<0){
                    return false;
                }
                sum+=a;
            }
            if(sum>zeroCount){
                return false;
            }
        }
        return true;
    }

    private boolean isS(int[] nums) {
        for (int i = 0; i < 4; i++) {
            if (nums[i] != nums[i + 1] - 1) {
                return false;
            }
        }
        return true;
    }
}
